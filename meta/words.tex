\section*{Słownik używanych skrótów:}

\begin{table}[H]
\begin{tabularx}{\textwidth}{ m{2.8cm} X }
	\textbf{BaaS} \dotfill & \ang{Backend~as~a~Service} --- udostępnianie jako usługi, gotowe~do~użycia komponenty serwerowe, np. mechanizm uwierzytelniania użytkowników \\
	\textbf{CSV} \dotfill & \ang{Comma-separated values} --- tekstowy format danych tabelarycznych rozdzielonych przecinkami~lub~średnikami \\
 	\textbf{DBSCAN} \dotfill & \ang{Density-Based Spatial Clustering of Applications with Noise} --- algorytm grupowania przestrzennego danych mogących zawierać szum \\
	\textbf{DOM} \dotfill & \ang{Document Object Model} --- standard reprezentacji dokumentów XML używany~przez~przeglądarki internetowe \\
	\textbf{DP} \dotfill & \ang{Device-independent pixel} --- jednostka wielkości niezależna~od~gęstości pikseli wyświetlacza, stosowana~przez~system Android \\
	\textbf{GPS} \dotfill & \ang{Global Positioning System} --- globalny system nawigacji satelitarnej stworzony~przez~rząd USA \\
	\textbf{JSON} \dotfill & \ang{JavaScript Object Notation} --- format zapisu obiektów wywodzący się~z~języka JavaScript \\
	\textbf{PRISMA} \dotfill & \ang{Preferred Reporting Items for Systematic Reviews and Meta-Analyses} ---~w~kontekście pracy diagram przedstawiający przepływ informacji~przez~kolejne kroki systematycznego przeglądu literatury \\
	\textbf{ROI} \dotfill & \ang{Return of Investment} --- wskaźnik zwrotu~z~inwestycji \\
	\textbf{UI} \dotfill & \ang{User interface} --- interfejs użytkownika \\
	\textbf{UX} \dotfill & \ang{User experience} --- wrażenia~a~także sposób interakcji użytkowników~z~produktem~i~jego doświadczania \\
\end{tabularx}
\end{table}
